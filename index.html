var fs = require('fs'); // Create an event listener that listens for the 'submit' event on the form document.querySelector('form').addEventListener('submit', function(e) { // Prevent the form from being submitted e.preventDefault(); // Get the visitor's name from the input box var name = document.querySelector('#name').value; // Save the name to a local CSV file named "names.csv" fs.appendFile('names.csv', name + '\n', function(err) { if (err) { console.error(err); } else { console.log('Name saved successfully!'); } }); });
